80double obliczOdleglosc(Punkt a, Punkt b){
81 return sqrt((b.x - a.x) * (b.x - a.x) + (b.y - a.y) * (b.y - a.y));
82}

95long orientacja(Punkt a, Punkt b, Punkt c){
96 long det = (b.y - a.y) * (c.x - b.x) - (b.x - a.x) * (c.y - b.y);
97 if(det == 0){
98 return 0; //liniowe
99 }
100 return (det > 0) ? 1 : 2; //1 jeśli pasuje i 2 jeśli nie pasuje
101}

167bool wyszukiwanieWszerz() {
168 queue<int> kolejka;
169 for (int u = 0; u < MAKS; ++u) {
170 if (skojarzeniaZPrzodu[u] == -1) {
171 odleglosc[u] = 0;
172 kolejka.push(u);
173 } else {
174 odleglosc[u] = INT_MAX;
175 }
176 }
177 odleglosc[-1] = INT_MAX;
178 while (!kolejka.empty()) {
179 int u = kolejka.front();
180 kolejka.pop();
181 if (odleglosc[u] < odleglosc[-1]) {
182 for (int v : listaSasiedztwa[u]) {
183 if (odleglosc[skojarzeniaZTylu[v]] == INT_MAX) {
184 odleglosc[skojarzeniaZTylu[v]] = odleglosc[u] + 1;
185 kolejka.push(skojarzeniaZTylu[v]);
186 }
187 }
188 }
189 }
190 return (odleglosc[-1] != INT_MAX);
191}

202bool wyszukiwanieWGlab( int u ) {
203 if (u != -1) {
204 for (int v : listaSasiedztwa[u]) {
205 if (odleglosc[skojarzeniaZTylu[v]] == odleglosc[u] + 1) {
206 if (wyszukiwanieWGlab(skojarzeniaZTylu[v])) {
207 skojarzeniaZTylu[v] = u;
208 skojarzeniaZPrzodu[u] = v;
209 return true;
210 }
211 }
212 }
213 odleglosc[u] = INT_MAX;
214 return false;
215 }
216 return true;
217}

226int algorytmHopcroftaKarpa() {
227 memset(skojarzeniaZPrzodu, -1, sizeof(skojarzeniaZPrzodu));
228 memset(skojarzeniaZTylu, -1, sizeof(skojarzeniaZTylu));
229 int skojarzenia = 0;
230 while (wyszukiwanieWszerz()) {
231 for (int u = 0; u < MAKS; ++u) {
232 if (skojarzeniaZPrzodu[u] == -1 && wyszukiwanieWGlab(u)) {
233 skojarzenia++;
234 }
235 }
236 }
237 return skojarzenia;
238}
