55std::string dekodujHuffman(const std::string& zakodowana_melodia, const std::string& kody){
56 std::unordered_map<std::string, char> mapa_kodow;
57 std::stringstream ss(kody);
58 std::string kod, znak;
59 while(ss >> kod >> znak){
60 mapa_kodow[kod] = znak[0];
61 }
62 std::stringstream odkodowana_melodia("");
63 kod = "";
64 for(char c : zakodowana_melodia){
65 kod += c;
66 if(mapa_kodow.find(kod) != mapa_kodow.end()){
67 if(mapa_kodow[kod] == '_'){
68 odkodowana_melodia << ' ';
69 }else{
70 odkodowana_melodia << mapa_kodow[kod];
71 }
72 kod = "";
73 }
74 }
75 return odkodowana_melodia.str();
76}
